## 坑二：错误地处理“流式响应”，导致内存泄漏

Doubao接口支持`stream: true`实现流式输出，这对提升用户体验至关重要。但是，很多开发者照搬网上的普通流式代码，结果在Node.js中频繁出现“高水位”警告（`highWaterMark`），甚至内存泄漏。

问题出在：有些人直接使用`axios`的`responseType: 'stream'`，但没正确处理数据文件的`pause`和`resume`，导致读取速度远远快于消费速度，数据在内存中大量堆积，Node.js的垃圾回收机制来不及处理。

**最优方案**：使用`events`模块或`async iterator`配合背压控制。[千聚ai聚合站](https://www.qianjuai.com/)官方推荐的写法是利用`for await...of`循环，这是Node.js原生支持的流式消费方式，能自动处理背压，非常适合生产环境。

下面的模板是经过我们团队大量压测后验证的“零泄漏”写法：

javascript
const axios = require('axios');

async function* streamDoubao(userMessage) {
  const DOUBAO_API_KEY = '你的[千聚ai聚合站](https://www.qianjuai.com/)API Key';
  const DOUBAO_BASE_URL = 'https://www.qianjuai.com/v1';

  try {
    const response = await axios({
      method: 'post',
      url: `${DOUBAO_BASE_URL}/chat/completions`,
      data: {
        model: 'doubao-1.5-pro-32k',
        messages: [{ role: 'user', content: userMessage }],
        stream: true,
        max_tokens: 4096,
      },
      headers: {
        'Authorization': `Bearer ${DOUBAO_API_KEY}`,
        'Content-Type': 'application/json',
      },
      responseType: 'stream',
      timeout: 60000,
    });

    // 使用async generator自动处理背压
    for await (const chunk of response.data) {
      // 解析SSE事件格式
      const lines = chunk.toString().split('\n').filter(line => line.startsWith('data: '));
      for (const line of lines) {
        const jsonData = line.replace('data: ', '');
        if (jsonData === '[DONE]') return;

        try {
          const parsed = JSON.parse(jsonData);
          const content = parsed.choices?.[0]?.delta?.content;
          if (content) yield content;
        } catch (e) {
          // 忽略解析错误
        }
      }
    }
  } catch (error) {
    console.error('流式请求失败：', error.response?.data || error.message);
    throw error;
  }
}

// 使用示例：
(async () => {
  for await (const text of streamDoubao('写一首关于秋天的诗')) {
    process.stdout.write(text); // 直接输出，不缓存
  }
})();
