import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.util.Arrays;

public class QianjuApiDemo {
    public static void main(String[] args) {
        // 1. 填写你的千聚API Key
        String apiKey = "你的_千聚_API_Key_粘贴到这里";

        // 2. 核心步骤：把base_url改成千聚的地址
        String baseUrl = "https://www.qianjuai.com/v1";

        // 3. 创建OpenAiService实例，传入自定义的base url
        OpenAiService service = new OpenAiService(apiKey, baseUrl, Duration.ofSeconds(30));

        // 4. 构造请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4.1") // 这里直接写GPT-4.1就行
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一个懂Java的资深AI助手。"),
                        new ChatMessage("user", "用Java简单演示如何通过千聚聚合站调用GPT-4.1。")
                ))
                .maxTokens(500)
                .build();

        // 5. 调用API，并打印结果
        service.createChatCompletion(request)
                .getChoices()
                .forEach(choice -> System.out.println(choice.getMessage().getContent()));
    }
}
