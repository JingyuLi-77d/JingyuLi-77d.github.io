## Grok 模型接入 Node.js，一行代码都不用改

既然千聚兼容 OpenAI 接口，那接 Grok 就跟接 GPT 一样简单。这里直接贴一个 Node.js 示例，三步搞定：

### 1. 安装 axios（或者用 openai 官方 SDK）

bash
npm install axios


### 2. 写一个 chatCompletion 函数

javascript
const axios = require('axios');

const API_KEY = 'sk-千聚申请的key'; // 替换成你的
const BASE_URL = 'https://www.qianjuai.com/v1';

async function chatWithGrok(messages) {
  try {
    const response = await axios.post(
      `${BASE_URL}/chat/completions`,
      {
        model: 'grok-1', // 千聚支持的 Grok 模型名称，以官方文档为准
        messages: messages,
        max_tokens: 1024,
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json',
        },
      }
    );
    return response.data.choices[0].message.content;
  } catch (error) {
    console.error('调用 Grok 失败', error.response?.data || error.message);
  }
}

// 使用示例
const messages = [
  { role: 'user', content: '用一句超短的话解释什么是Grok模型' },
];
chatWithGrok(messages).then(console.log);


### 3. 跑起来

bash
node index.js
