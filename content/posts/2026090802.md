来看核心代码，工具就用OkHttp，简单粗暴：

java
import okhttp3.*;
import org.json.JSONObject;
import java.io.IOException;

public class MistralLargeCaller {

    // 核心：只要改这俩参数
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    private static final String API_KEY = "你的千聚API_KEY";

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        // 构建请求体（OpenAI兼容格式）
        JSONObject jsonBody = new JSONObject();
        jsonBody.put("model", "mistralai/Mixtral-8x22B-Instruct-v0.1"); // 指定模型
        jsonBody.put("max_tokens", 1024);
        jsonBody.put("temperature", 0.7);

        // Messages 部分
        JSONObject systemMessage = new JSONObject();
        systemMessage.put("role", "system");
        systemMessage.put("content", "你是一个专业的Java开发顾问。");
        JSONObject userMessage = new JSONObject();
        userMessage.put("role", "user");
        userMessage.put("content", "请用中文详细解释如何利用Java调用Mistral模型。");

        jsonBody.put("messages", new JSONObject[]{systemMessage, userMessage});

        // 构建HTTP请求
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(jsonBody.toString(), MediaType.get("application/json; charset=utf-8")))
                .build();

        // 同步调用（生产环境建议用异步或流式）
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseString = response.body().string();
                JSONObject respJson = new JSONObject(responseString);
                String answer = respJson.getJSONArray("choices")
                        .getJSONObject(0)
                        .getJSONObject("message")
                        .getString("content");
                System.out.println("MistralLarge回答：\n" + answer);
            } else {
                System.err.println("请求失败: " + response.code());
            }
        }
    }
}
