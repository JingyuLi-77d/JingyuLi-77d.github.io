import com.theokanning.openai.OpenAiService;

import com.theokanning.openai.completion.chat.ChatMessage;

// 关键点1：替换baseUrl

// 关键点2：替换为你从千聚获取的apiKey

OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(60));
service.setBaseUrl(baseUrl); // 指向千聚API

// 构建一个简单的聊天请求
ChatCompletionRequest request = ChatCompletionRequest.builder()
    .model("gemini-3-pro")  // 直接指定模型名字
    .messages(
        List.of(
            new ChatMessage("system", "你是一个乐于助人的助手。"),
            new ChatMessage("user", "用一句话解释低代码平台。")
        )
    )
    .maxTokens(100)
    .build();

// 发送请求，获取回复
service.createChatCompletion(request).getChoices().forEach(
    choice -> System.out.println(choice.getMessage().getContent())
);
