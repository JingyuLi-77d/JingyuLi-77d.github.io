## 接入到底有多简单 —— Java示例

为了让所有人看懂，我写了一个直接用Java接入Qwen-Plus的示例。没有Spring Boot、没有WebFlux、没有复杂代理依赖——纯Java + 原生HTTP，跑起来零额外配置。

java
import okhttp3.*;
import com.alibaba.fastjson2.*;

public class demo {
    public static void main(String[] args) {

        String apiKey = "sk-your-key";  // 替换成千聚申请来的key

        OkHttpClient client = new OkHttpClient().newBuilder()
                .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        // 构造请求体
        JSONObject body = new JSONObject();
        body.put("model", "gpt-4o-mini");  // 兼容模式，可以是OpenAI、Qwen等任意模型
        JSONArray messages = new JSONArray();
        JSONObject msg = new JSONObject();
        msg.put("role", "user");
        msg.put("content", "请用Java写一段冒泡排序代码");
        messages.add(msg);
        body.put("messages", messages);
        body.put("stream", false); // 如果流式输出就设true

        Request request = new Request.Builder()
                // 真正关键的改动只在这一行
                .url("https://www.qianjuai.com/v1/chat/completions")
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        body.toJSONString()
                ))
                .addHeader("Authorization", "Bearer " + apiKey)
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String resBody = response.body().string();
                System.out.println(resBody);
            } else {
                System.out.println("Error: " + response.code());
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
