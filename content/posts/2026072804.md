# 调用百度ERNIE-Bot模型（以ERNIE-Bot为例）
def call_ernie(prompt):
    try:
        response = client.chat.completions.create(
            model="ernie-bot",   # 千聚已兼容ERNIE标准模型名
            messages=[
                {"role": "user", "content": prompt}
            ],
            stream=False,        # 改为True可体验流式输出
            temperature=0.7,
            max_tokens=1000
        )
        return response.choices[0].message.content
    except Exception as e:
        return f"调用出错：{e}"

# 测试对话
result = call_ernie("用一句话解释什么是人工智能")
print(result)

# 多轮对话示例
def multi_turn_ernie():
    conversation = [
        {"role": "user", "content": "你好，ERNIE！"}
    ]
    
    response = client.chat.completions.create(
        model="ernie-bot",
        messages=conversation,
        stream=False,
        max_tokens=500
    )
    
    assistant_reply = response.choices[0].message.content
    print(f"第一轮回复：{assistant_reply}")
    
    # 继续第二轮对话
    conversation.append({"role": "assistant", "content": assistant_reply})
    conversation.append({"role": "user", "content": "请告诉我ERNIE能应用在哪些场景"})
    
    response2 = client.chat.completions.create(
        model="ernie-bot",
        messages=conversation,
        stream=False,
        max_tokens=800
    )
    
    print(f"第二轮回复：{response2.choices[0].message.content}")

multi_turn_ernie()
