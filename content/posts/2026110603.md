### 完整的Java示例代码（关键部分）

下面是一个高度概括、不依赖外部jar包、可直接在Java 8+环境运行的类。去掉冗余注释，保留最核心的逻辑。

java
package com.example.demo.ai;

import java.io.*;
import java.net.HttpURLConnection;
import java.net.URL;
import java.nio.charset.StandardCharsets;

public class AIDemo {

    // 这个Key在[千聚ai大模型中转站](https://www.qianjuai.com/)后台获取
    private static final String API_KEY = "sk-your-key-here"; 
    // 统一的API网关
    private static final String API_BASE_URL = "https://www.qianjuai.com/v1/chat/completions";

    public static String callModel(String modelName, String userPrompt) {
        try {
            URL url = new URL(API_BASE_URL);
            HttpURLConnection connection = (HttpURLConnection) url.openConnection();
            connection.setRequestMethod("POST");
            connection.setRequestProperty("Content-Type", "application/json");
            connection.setRequestProperty("Authorization", "Bearer " + API_KEY);
            connection.setDoOutput(true);

            // 构建标准OpenAI格式的JSON请求体
            String jsonInput = "{" 
                    + "\"model\": \"" + modelName + "\","
                    + "\"messages\": [{\"role\": \"user\", \"content\": \"" + userPrompt + "\"}],"
                    + "\"max_tokens\": 1024,"
                    + "\"temperature\": 0.7"
                    + "}";

            try (OutputStream os = connection.getOutputStream()) {
                byte[] input = jsonInput.getBytes(StandardCharsets.UTF_8);
                os.write(input, 0, input.length);
            }

            // 获取响应
            int responseCode = connection.getResponseCode();
            BufferedReader in = new BufferedReader(
                    new InputStreamReader(connection.getInputStream(), StandardCharsets.UTF_8));
            String inputLine;
            StringBuilder response = new StringBuilder();
            while ((inputLine = in.readLine()) != null) {
                response.append(inputLine);
            }
            in.close();
            return response.toString();
            
        } catch (Exception e) {
            e.printStackTrace();
            return "Error occurred: " + e.getMessage();
        }
    }

    // 测试示例：
    public static void main(String[] args) {
        // 换这3个名字，就调通了3大平台！
        System.out.println("=== 调用 OpenAI GPT-4o ===");
        System.out.println(callModel("gpt-4o", "用中文简单介绍下你自己"));

        System.out.println("=== 调用 DeepSeek ===");
        System.out.println(callModel("deepseek-chat", "讲个冷笑话"));

        System.out.println("=== 调用 通义千问 Qwen ===");
        System.out.println(callModel("qwen-turbo", "现在几点钟？"));
    }
}
