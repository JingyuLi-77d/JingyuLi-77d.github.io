import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.completion.chat.ChatMessageRole;
import com.theokanning.openai.service.OpenAiService;
import java.util.Arrays;
import java.util.List;

public class Gemini3ProDemo {

    public static void main(String[] args) {
        // 第一步：替换baseUrl为千聚API的地址
        String baseUrl = "https://www.qianjuai.com/v1";
        // 第二步：将apiKey替换为你从千聚平台获取的Key
        String apiKey = "your_qianju_api_key";

        // 创建OpenAiService实例，传参时指定baseUrl
        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(60), baseUrl);

        // 第三步：构建请求，这里的model名称直接使用 "gemini-3-pro" 或千聚标注的模型别名
        List<ChatMessage> messages = Arrays.asList(
            new ChatMessage(ChatMessageRole.SYSTEM.value(), "你是一个专业的Java开发和AI助手。"),
            new ChatMessage(ChatMessageRole.USER.value(), "请用Java写一个使用Gemini 3 Pro API进行多轮对话的示例程序。")
        );

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gemini-3-pro") // 注意：请确认千聚平台上的具体模型ID
                .messages(messages)
                .maxTokens(2048)
                .temperature(0.8)
                .build();

        String response = service.createChatCompletion(request)
                .getChoices()
                .get(0)
                .getMessage()
                .getContent();

        System.out.println("Gemini 3 Pro Response:");
        System.out.println(response);

        // 关闭服务（可选，但推荐）
        // service.shutdownExecutor();
    }
}
